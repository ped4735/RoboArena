tree("Root")
	parallel
		repeat mute tree "Follow"
		repeat mute tree "Dash"

tree("Follow")
	while not isDashing
		fallback
			while not ReachTargetAtDistance
				sequence
					RotateToFaceTarget
					FollowTargetRb
			while not isLookToTarget
					sequence
						Stop
						RotateToFaceTarget
			while isLookToTarget
				Stop

tree("Dash")
	fallback
		while ReachTargetAtDistance
			while isLookToTarget
				sequence
					Wait 1.0
					Dash2(true)					
		while not ReachTargetAtDistance
			Dash2(false)
