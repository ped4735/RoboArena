tree("Root")
	parallel
		repeat mute tree "Follow"
		repeat mute tree "Shoot"

tree("Follow")
	fallback
		while not ReachTargetAtDistance()
			sequence
				RotateToFaceTarget
				FollowTargetT
		while not isLookToTarget
				RotateToFaceTarget

tree("Shoot")
	fallback
		while ReachTargetAtDistance()
				while not isLookAtObstacle()
					sequence
						Shoot
						WaitAttackDelay
